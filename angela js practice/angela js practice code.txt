function fabonacci(number){
  var output = [];
  if (number===1){ 
    output = [0];
  }
  else if (number===2){
    output = [0,1];
  }
  else {
    output = [0,1];
    for (var i = 2; i < number; i++){
      output.push(output[i - 2] + output[i - 1])
    }
  }
    
  
console.log(output);
  return output;
}
